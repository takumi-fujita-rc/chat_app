## データ型定義

```typescript
// src/types/chat.ts

type Role = "user" | "assistant";

interface Message {
  id: string;          // ユニークID（crypto.randomUUID()）
  role: Role;          // 送信者
  content: string;     // メッセージ本文
  createdAt: Date;     // 作成日時
}

// API リクエスト
interface ChatRequest {
  messages: Pick<Message, "role" | "content">[];
}

// API レスポンス
interface ChatResponse {
  message: string;     // AIの返答テキスト
}
```
